import pandas as pd
import time

from sklearn.model_selection import train_test_split
from sklearn.feature_extraction.text import TfidfVectorizer
from sklearn.linear_model import LogisticRegression
from sklearn.ensemble import RandomForestClassifier
from sklearn.metrics import accuracy_score

df = pd.read_csv("dataset.csv")

X_train, X_test, y_train, y_test = train_test_split(
    df["text"], df["label"], test_size=0.2, random_state=42
)

vectorizer = TfidfVectorizer()
X_train_vec = vectorizer.fit_transform(X_train)
X_test_vec = vectorizer.transform(X_test)

# Logistic Regression
start = time.time()
lr = LogisticRegression()
lr.fit(X_train_vec, y_train)
lr_preds = lr.predict(X_test_vec)
lr_time = time.time() - start

# Random Forest
start = time.time()
rf = RandomForestClassifier()
rf.fit(X_train_vec, y_train)
rf_preds = rf.predict(X_test_vec)
rf_time = time.time() - start

print("Logistic Accuracy:", accuracy_score(y_test, lr_preds))
print("RF Accuracy:", accuracy_score(y_test, rf_preds))
print("LR Time:", lr_time)
print("RF Time:", rf_time)
